import java.util.ArrayList;
import java.util.List;

    public static void main(String[] args) {
        // 產生一個List
        List<Pricing> list = new ArrayList();

        // 加入幾個[學校訂價]物件

        list.add(new SchoolPricing("記事本", 20000, 65000, 850));
