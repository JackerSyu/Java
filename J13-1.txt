package com.abc;

import javax.swing.*;
import java.awt.*;

public class MyFrame extends JFrame {
    public MyFrame(){        
        initUI();
    }
    
    private void initUI(){
        this.setTitle("我的JFrame");
        this.getContentPane().setBackground(new Color(100, 100, 100));
        this.setSize(new Dimension(300,300));
        this.setVisible(true);        
    }
}
